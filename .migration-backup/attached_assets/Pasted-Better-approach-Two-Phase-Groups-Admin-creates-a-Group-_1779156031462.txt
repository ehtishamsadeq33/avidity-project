PART 4 — Update Survey verify-ic endpoint:
typescriptrouter.post('/survey/verify-code', async (req, res) => {
  const { code, icNumber } = req.body;
  if (!code || !icNumber) {
    return res.status(400).json({ error: 'Code and IC number are required' });
  }

  // Find group by PRE or POST code
  const group = await Group.findOne({
    $or: [{ preCode: code }, { postCode: code }]
  });

  if (!group) {
    return res.status(404).json({ error: 'Invalid code. Please check and try again.' });
  }

  // Determine phase
  const phase = group.preCode === code ? 'pre' : 'post';

  // Check if phase is enabled
  if (phase === 'pre' && !group.preEnabled) {
    return res.status(400).json({ error: 'PRE survey is currently closed for this group.' });
  }
  if (phase === 'post' && !group.postEnabled) {
    return res.status(400).json({ error: 'POST survey is currently closed for this group.' });
  }

  // Check if IC number is in the group
  if (!group.icNumbers.includes(icNumber.trim())) {
    return res.status(403).json({ error: 'Your IC number is not registered in this group. Please contact your administrator.' });
  }

  // Check if already submitted for this phase
  const existing = await Submission.findOne({
    icNumber: icNumber.trim(),
    groupId: group._id,
    phase,
  });
  if (existing) {
    return res.status(400).json({
      error: `You have already completed the ${phase.toUpperCase()} survey for this group.`
    });
  }

  res.json({
    verified: true,
    groupId: group._id,
    groupName: group.groupName,
    phase,
    message: `Verified! You are entering the ${phase.toUpperCase()} survey for ${group.groupName}`
  });
});
