PART 2 — Implement PDF download that captures all 3 report pages:

Install react-native-view-shot and expo-print and expo-sharing
In ReportPage3, implement the downloadPDF function:

javascriptimport { captureRef } from 'react-native-view-shot';
import * as Print from 'expo-print';
import * as Sharing from 'expo-sharing';

const page1Ref = useRef(null);
const page2Ref = useRef(null);
const page3Ref = useRef(null);

const downloadPDF = async () => {
  try
  {
    // Capture all 3 pages as base64 images
    const page1Image = await captureRef(page1Ref, { format: 'png', quality: 1 });
    const page2Image = await captureRef(page2Ref, { format: 'png', quality: 1 });
    const page3Image = await captureRef(page3Ref, { format: 'png', quality: 1 });

    // Convert to base64
    const toBase64 = async (uri) => {
      const response = await fetch(uri);
      const blob = await response.blob();
      return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onloadend = () => resolve(reader.result);
        reader.readAsDataURL(blob);
      });
    };

    const img1 = await toBase64(page1Image);
    const img2 = await toBase64(page2Image);
    const img3 = await toBase64(page3Image);

    // Build HTML with all 3 pages
    const html = `
      <html>
        <body style="margin:0;padding:0;background:#fffde7;">
          <div style="page-break-after:always;">
            <img src="${img1}" style="width:100%;" />
          </div>
          <div style="page-break-after:always;">
            <img src="${img2}" style="width:100%;" />
          </div>
          <div>
            <img src="${img3}" style="width:100%;" />
          </div>
        </body>
      </html>
    `;

    // Generate PDF
    const { uri } = await Print.printToFileAsync({ html, base64: false });

    // Share/download the PDF
    await Sharing.shareAsync(uri, {
      mimeType: 'application/pdf',
      dialogTitle: 'Download CRA Report',
      UTI: 'com.adobe.pdf'
    });
  } catch (e) {
    Alert.alert('Error', 'Could not generate PDF. Please try again.');
  }
};
