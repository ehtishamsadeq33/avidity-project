PART 2 — Backend: New group average endpoint:
Add this route to admin routes:
typescript// Get group averages for Skills, Will, Environmental Support
router.get('/groups/:id/averages', requireAdmin, async (req, res) => {
  try {
    const group = await Group.findById(req.params.id);
    if (!group) return res.status(404).json({ error: 'Group not found' });

    const submissions = await Submission.find({ groupCode: group.groupCode });
    const total = submissions.length;

    if (total === 0) {
      return res.json({
        total: 0,
        groupSkills: null,
        groupWill: null,
        groupEnvironmentalSupport: null,
        candidates: [],
      });
    }

    // Sum up each dimension across all candidates
    let sumSkills = 0;
    let sumWill = 0;
    let sumEnv = 0;

    const candidates = submissions.map(s => ({
      name: s.name,
      skills: s.skillsScore,       // use your actual field name
      will: s.willScore,           // use your actual field name
      environmentalSupport: s.environmentalSupportScore, // use your actual field name
    }));

    candidates.forEach(c => {
      sumSkills += c.skills ?? 0;
      sumWill += c.will ?? 0;
      sumEnv += c.environmentalSupport ?? 0;
    });

    res.json({
      total,
      groupSkills: parseFloat((sumSkills / total).toFixed(1)),
      groupWill: parseFloat((sumWill / total).toFixed(1)),
      groupEnvironmentalSupport: parseFloat((sumEnv / total).toFixed(1)),
      candidates,
    });
  } catch (err) {
    res.status(500).json({ error: 'Failed to calculate group averages' });
  }
});
