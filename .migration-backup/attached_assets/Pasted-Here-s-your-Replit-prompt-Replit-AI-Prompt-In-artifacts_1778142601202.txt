  if (total <= 0) return;
  var cx = 80, cy = 80, radius = 75;
  var startAngle = -Math.PI / 2;
  entries.forEach(function(e, i) {
    var count = Number(e[1]) || 0;
    if (count <= 0) return;

    ctx.beginPath();
    ctx.moveTo(cx, cy);
    ctx.arc(cx, cy, radius, startAngle, startAngle + sliceAngle);
    ctx.closePath();
    ctx.fillStyle = colors[i] || '#ccc';
    ctx.fill();
    startAngle += sliceAngle;
  });
