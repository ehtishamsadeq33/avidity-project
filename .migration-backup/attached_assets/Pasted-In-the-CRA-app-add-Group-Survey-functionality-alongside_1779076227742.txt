  const submissions = await Submission.find({

typescriptrouter.post('/survey/verify-ic', async (req, res) => {
  const { icNumber } = req.body;
  const group = await Group.findOne({ icNumbers: icNumber });
  if (!group) {
    return res.status(404).json({ error: 'IC number not found in any group. Please contact your administrator.' });
  }
  // Check if already submitted
  const existing = await Submission.findOne({ icNumber, groupId: group._id });
  if (existing) {
    return res.status(400).json({ error: 'You have already submitted this survey.' });
  }
  res.json({ verified: true, groupId: group._id, groupName: group.groupName });
});

PART 8 — IC Number entry screen (app/ic-entry.tsx):
javascriptexport default function ICEntryScreen() {
  const [icNumber, setIcNumber] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const router = useRouter();
  const params = useLocalSearchParams();
  const isGroup = params.type === 'group';

  const handleContinue = async () => {
    if (!icNumber.trim()) {
      setError('Please enter your IC number');
      return;
    }
    if (isGroup) {
      setLoading(true);
      try {
        const res = await fetch(`${API_BASE}/survey/verify-ic`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ icNumber: icNumber.trim() }),
        });
        const data = await res.json();
        if (!res.ok) {
          setError(data.error ?? 'Verification failed');
          return;
        }
        router.push({
          pathname: '/personal',
          params: { icNumber: icNumber.trim(), groupId: data.groupId, type: 'group' }
        });
      } catch (e) {
        setError('Network error. Please try again.');
      } finally {
        setLoading(false);
      }
    } else {
      router.push({
        pathname: '/personal',
        params: { icNumber: icNumber.trim(), type: 'individual' }
      });
    }
  };

  return (
    <View style={{ flex: 1, backgroundColor: '#fffde7', padding: 24, justifyContent: 'center' }}>
      <Text style={{ fontSize: 22, fontWeight: 'bold', marginBottom: 8, color: '#1A1A1A' }}>
        {isGroup ? 'Group Survey' : 'Individual Survey'}
      </Text>
      <Text style={{ fontSize: 14, color: '#555', marginBottom: 24 }}>
        Please enter your IC number to continue
      </Text>
      <TextInput
        style={{
          borderWidth: 1, borderColor: '#ccc', borderRadius: 10,
          padding: 14, fontSize: 16, backgroundColor: '#fff', marginBottom: 12
        }}
        placeholder="IC Number (e.g. 901234-56-7890)"
        value={icNumber}
        onChangeText={setIcNumber}
        autoCapitalize="none"
      />
      {error ? <Text style={{ color: 'red', marginBottom: 12 }}>{error}</Text> : null}
      <TouchableOpacity
        onPress={handleContinue}
        disabled={loading}
        style={{
          backgroundColor: '#f5c842', borderRadius: 10, padding: 14, alignItems: 'center'
        }}
      >
        <Text style={{ fontWeight: 'bold', fontSize: 16, color: '#000' }}>
          {loading ? 'Verifying...' : 'Continue →'}
        </Text>
      </TouchableOpacity>
    </View>
  );
}
