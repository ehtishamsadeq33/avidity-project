PART 5 — Group Entry Screen (app/group-entry.tsx):
javascriptexport default function GroupEntryScreen() {
  const [groupCode, setGroupCode] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const router = useRouter();

  const handleContinue = async () => {
    if (!groupCode.trim()) {
      setError('Please enter a group code');
      return;
    }
    setLoading(true);
    setError('');
    try {
      const res = await fetch(`${API_BASE}/survey/verify-group-code`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ groupCode: groupCode.trim().toUpperCase() }),
      });
      const data = await res.json();
      if (!res.ok) {
        setError(data.error ?? 'Invalid group code');
        return;
      }
      router.push({
        pathname: '/personal',
        params: {
          groupCode: data.groupCode,
          groupId: data.groupId,
          groupName: data.groupName,
          submissionType: 'group',
        }
      });
    } catch (e) {
      setError('Network error. Please try again.');
    } finally {
      setLoading(false);
    }
  };

  return (
    <View style={{ flex: 1, backgroundColor: '#fffde7', padding: 24, justifyContent: 'center' }}>
      <Text style={{ fontSize: 22, fontWeight: 'bold', marginBottom: 8, color: '#1A1A1A' }}>
        Group Survey
      </Text>
      <Text style={{ fontSize: 14, color: '#555', marginBottom: 24 }}>
        Enter the group code provided by your administrator
      </Text>
      <TextInput
        style={{
          borderWidth: 1, borderColor: '#ccc', borderRadius: 10,
          padding: 14, fontSize: 16, backgroundColor: '#fff',
          marginBottom: 12, textTransform: 'uppercase', letterSpacing: 2,
        }}
        placeholder="e.g. ABC123"
        value={groupCode}
        onChangeText={(t) => setGroupCode(t.toUpperCase())}
        autoCapitalize="characters"
        autoCorrect={false}
      />
      {error ? <Text style={{ color: 'red', marginBottom: 12 }}>{error}</Text> : null}
      <TouchableOpacity
        onPress={handleContinue}
        disabled={loading}
        style={{ backgroundColor: '#f5c842', borderRadius: 10, padding: 14, alignItems: 'center' }}
      >
        <Text style={{ fontWeight: 'bold', fontSize: 16, color: '#000' }}>
          {loading ? 'Verifying...' : 'Continue →'}
        </Text>
      </TouchableOpacity>
    </View>
  );
}
